C Programlama Fonksiyonlar

Fonksiyon Tanımlama ve Kullanma



Void Fonksiyonu



#include <stdio.h>
void fonksiyon(int number , char karakter){
printf("number : %d karakter : %c",number,karakter);
}
int main(){
int sayi = 10;
char harf = 'a';
fonksiyon(sayi,harf);
return 0;
}

Veri Döndüren Fonksiyonlar




#include <stdio.h>
int fonksiyon(int number , int number1){
int y = number + number1;
return y;
}
int main(){
int sayi = 10 , sayi1 = 20;
int k = fonksiyon(sayi,sayi1);
printf("Fonksiyondan döndürülen sayı : %d ",k);
return 0;
}
