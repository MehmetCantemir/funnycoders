Değişken Tanımlama ve Ekrana Yazdırma


Ekran Çıktısı :


#include <stdio.h>
int main(){
int sayi = 10;
double sayi1 = 10.5;
float sayi2 = 17.5;
char karakter = 'a';
printf("\n Integer : %d",sayi);
printf("\n Double : %lf",sayi1);
printf("\n Float : %.2f",sayi2);
printf("\n* Char : %.2c",karakter);
printf("Hello World");
return 0;
}


Kullanıcıdan Veri Alma (scanf)


Ekran Çıktısı:


#include <stdio.h>
int main(){
int number;
printf("Bir sayı giriniz : ");
scanf("%d",&number);
printf("Girilen Sayı : %d",number);
return 0;
}

//Çarpma işlemi
printf("x ve y'nin çarpımı = %d",x*y);
//Bölme işlemi
printf("x ve y'nin bölümü = %d",x/y);
