strcmp Fonksiyonu




#include <stdio.h>
#include <string.h>
int main(){
char name[] = "mehmet";
char surname[] = "mehmet1";
int r;
r=strcmp(name,surname);
if printf("\n name karakter dizisi surname karakter dizisinden küçüktür");
else if printf("\n name karakter dizisi surname karakter dizisinden büyüktür");
else printf("\n name karakter dizisi surname karakter dizisi eşittir");

}
