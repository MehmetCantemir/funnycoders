Dizilere Değer Atama

Index numaralarına göre değer atama



#include <stdio.h>
int main(){
int dizi [20];
dizi [0]=1;
dizi [1]=2;
dizi [20]=15;
printf("dizi[0] : %d dizi[1]: %d dizi[20] : %d",dizi[0],dizi[1],dizi[20]);
return 0;
}

Parantez İçinde Değer Atama




#include <stdio.h>
int main(){
int dizi[20] = {1,2,3,4,5,6,7};
printf("dizi[0] : %d dizi[1] : %d dizi[3] : %d dizi[4] : %d dizi[5] : %d",dizi[0],dizi[1],dizi[2],dizi[3],dizi[4]);
return 0;
}

Döngü ile Değer Atama



#include <stdio.h>
int main(){
int dizi [20];
for (int i=0 ; i < 20 ; i++)
{
printf(" %d ",dizi[i]);
}
return 0;
}
